          git init - zainicjalizuj nowe lokalne repozytorium
          git status - status repozytorium
          git add - dodaj do sledzenia
          git add . - dodaj do sledzenia wszystkie pliki
          
          git commit - dodaj commit ( odpala vim )
          git commit -m "..." - dodaj commit o tresci ...
          
          // Dodanie zdalnego repozytorium
          git remote add origin git@github.com:oskarski/ALX_30_oct.git
          
          // Zmiana linku do zdalnego repozytorium
          git remote set-url origin git@github.com:oskarski/ALX_30_oct.git
          
          git push -u origin main
          git push -> kiedy juz mamy powiazanie z origin main
          git clone -> slklonuj repozytorium na lokalna maszyne
          git log -> historia commitow
                q -> wyjscie ( jesli duza historia )

          git reset --hard <- zresetuj do ostatniego commita

          UWAGA
          git reset --hard HEAD~2
          git push origin main --force 

          git branch -> lista galezi ( branche )
          git checkout -b feature -> ( stworz branch feature i na niego przejdz )
          git checkout feat -> przejdz na brrancha feat
          git branch feat -> nowy branch feat

          git pull origin main -> pobranie zmian z repozytorium zdalnego z brancha main
